18 /**
19  * @brief Example that shows how to use the motion detector
20  *
21  * This is an example of how the motion detector can be used.
22  * The example executes as follows:
23  * - Activate Radar System Software (RSS)
24  * - Set log level, optional
25  * - Create a motion detector configuration
26  * - Create a motion detector using the previously created configuration
27  * - Activate the motion detector
28  * - Get the result and print it 5 times
29  * - Deactivate and destroy the motion detector
30  * - Reconfigure the motion detector
31  * - Activate the motion detector
32  * - Get the result and print it 5 times
33  * - Deactivate and destroy the motion detector
34  * - Destroy the motion detector configuration
35  * - Deactivate Radar System Software
36  */

99 bool execute_motion_detection(acc_detector_motion_configuration_t motion_configuration)
100 {
101  bool success = true;
102  acc_detector_motion_handle_t handle = acc_detector_motion_create(motion_configuration);
103  float update_rate = acc_detector_motion_configuration_update_rate_get(motion_configuration);
104 
105  if (handle != NULL)
106  {
107  if (acc_detector_motion_activate(handle))
108  {
109  bool motion_detect = false;
110  uint_fast8_t sweep_count = 5;
111 
112  while (sweep_count > 0)
113  {
114  uint32_t start_time_us;
115  hal.os.gettime(&start_time_us);
116 
117  if (acc_detector_motion_get_next(handle, &motion_detect))
118  {
119  printf("Motion detect: %s\n", motion_detect ? "true" : "false");
120  }
121  else
122  {
123  printf("Detector motion data not properly retrieved\n");
124  success = false;
125  break;
126  }
127 
128  uint32_t stop_time_us;
129  hal.os.gettime(&stop_time_us);
130 
131  uint32_t sleep_us = 0;
132  uint_fast32_t update_period_us = (1.0f / update_rate) * 1000000.0f;
133  uint_fast32_t duration_us = stop_time_us - start_time_us;
134 
135  if (update_period_us > duration_us)
136  {
137  sleep_us = update_period_us - duration_us;
138  }
139 
140  hal.os.sleep_us(sleep_us);
141 
142  sweep_count--;
143  }
144 
145  if (!acc_detector_motion_deactivate(handle))
146  {
147  printf("acc_detector_motion_deactivate() failed\n");
148  success = false;
149  }
150  }
151  else
152  {
153  printf("acc_detector_motion_activate() failed\n");
154  success = false;
155  }
156  }
157  else
158  {
159  printf("acc_detector_motion_create() failed\n");
160  success = false;
161  }
162 
163  acc_detector_motion_destroy(&handle);
164 
165  return success;
166 }
