19 /**
20  * @brief Example that shows how to use the Power Bins service
21  *
22  * This is an example on how the Power Bins service can be used.
23  * The example executes as follows:
24  * - Activate Radar System Software (RSS)
25  * - Create a Power Bins service configuration (with blocking mode as default)
26  * - Create a Power Bins service using the previously created configuration
27  * - Activate the Power Bins service
28  * - Get the result and print it 5 times, where the last result is intentionally late
29  * - Deactivate and destroy the Power Bins service
30  * - Create and activate Power Bins service
31  * - Destroy the Power Bins service configuration
32  * - Deactivate Radar System Software
33  */

91 acc_service_status_t execute_power_bins_with_blocking_calls(acc_service_configuration_t power_bins_configuration)
92 {
93  acc_service_handle_t handle = acc_service_create(power_bins_configuration);
94 
95  if (handle == NULL)
96  {
97  printf("acc_service_create failed\n");
98  return ACC_SERVICE_STATUS_FAILURE_UNSPECIFIED;
99  }
100 
101  acc_service_power_bins_metadata_t power_bins_metadata;
102  acc_service_power_bins_get_metadata(handle, &power_bins_metadata);
103 
104  printf("Actual start: %u mm\n", (unsigned int)(power_bins_metadata.actual_start_m * 1000.0f));
105  printf("Actual length: %u mm\n", (unsigned int)(power_bins_metadata.actual_length_m * 1000.0f));
106  printf("Actual end: %u mm\n", (unsigned int)((power_bins_metadata.actual_start_m + power_bins_metadata.actual_length_m) * 1000.0f));
107  printf("Bin count: %u\n", (unsigned int)(power_bins_metadata.actual_bin_count));
108 
109  float data[power_bins_metadata.actual_bin_count];
110 
111  acc_service_power_bins_result_info_t result_info;
112  acc_service_status_t service_status = acc_service_activate(handle);
113 
114  if (service_status == ACC_SERVICE_STATUS_OK)
115  {
116  uint_fast8_t sweep_count = 5;
117 
118  while (sweep_count > 0)
119  {
120  service_status = acc_service_power_bins_get_next(handle, data, power_bins_metadata.actual_bin_count, &result_info);
121 
122  if (service_status == ACC_SERVICE_STATUS_OK)
123  {
124  printf("Power_bins result_info.sequence_number: %u\n", (unsigned int)result_info.sequence_number);
125  printf("Power_bins data:\n");
126  for (uint_fast16_t index = 0; index < power_bins_metadata.actual_bin_count; index++)
127  {
128  printf("%u\t", (unsigned int)(data[index]));
129  }
130 
131  printf("\n");
132  }
133  else
134  {
135  printf("Power_bins data not properly retrieved\n");
136  }
137 
138  sweep_count--;
139  }
140 
141  service_status = acc_service_deactivate(handle);
142  }
143  else
144  {
145  printf("acc_service_activate() %u => %s\n", (unsigned int)service_status, acc_service_status_name_get(service_status));
146  }
147 
148  acc_service_destroy(&handle);
149 
150  return service_status;
151 }
