16 /**
17  * @brief Example that shows how to use the distance peak detector
18  *
19  * This is an example on how the distance peak detector can be used.
20  * The example executes as follows:
21  * - Activate Radar System Software (RSS)
22  * - Create a distance peak detector configuration (with blocking mode as default)
23  * - Create a distance peak detector using the previously created configuration
24  * - Activate the distance peak detector
25  * - Get the result and print it 100 times
26  * - Deactivate and destroy the distance peak detector
27  * - Destroy the distance peak detector configuration
28  * - Deactivate Radar System Software
29  */

143  acc_detector_distance_peak_configuration_destroy(&distance_configuration);

149 acc_detector_distance_peak_status_t detect_with_blocking_calls(acc_detector_distance_peak_handle_t handle)
150 {
151  acc_detector_distance_peak_status_t detector_status = ACC_DETECTOR_DISTANCE_PEAK_STATUS_SUCCESS;
152  acc_detector_distance_peak_metadata_t metadata;
153  uint_fast16_t reflection_count = 10;
154  acc_detector_distance_peak_reflection_t reflections[reflection_count];
155 
156  if (handle == NULL)
157  {
158  fprintf(stderr, "handle == NULL\n");
159  return ACC_DETECTOR_DISTANCE_PEAK_STATUS_FAILURE;
160  }
161 
162  printf("Running distance peak detector in blocking mode\n");
163 
164  acc_detector_distance_peak_get_metadata(handle, &metadata);
165  printf("Actual start: %u mm\n", (unsigned int)(metadata.actual_start_m * 1000.0f));
166  printf("Actual length: %u mm\n", (unsigned int)(metadata.actual_length_m * 1000.0f));
167  printf("Actual end: %u mm\n", (unsigned int)((metadata.actual_start_m + metadata.actual_length_m) * 1000.0f));
168  printf("\n");
169 
170  acc_detector_distance_peak_result_info_t result_info;
171 
172  detector_status = acc_detector_distance_peak_activate(handle);
173 
174  if (detector_status != ACC_DETECTOR_DISTANCE_PEAK_STATUS_SUCCESS)
175  {
176  fprintf(stderr, "acc_detector_distance_peak_activate failed\n");
177  return ACC_DETECTOR_DISTANCE_PEAK_STATUS_FAILURE;
178  }
179 
180  for (int i = 0; i < ITERATIONS; i++)
181  {
182  reflection_count = 10;
183 
184  detector_status = acc_detector_distance_peak_get_next(handle,
185  reflections,
186  &reflection_count,
187  &result_info);
188 
189  if (detector_status == ACC_DETECTOR_DISTANCE_PEAK_STATUS_SUCCESS)
190  {
191  float start_m = metadata.actual_start_m;
192  float end_m = metadata.actual_start_m + metadata.actual_length_m;
193 
194  printf("Distance detector: Reflections: %u. Seq. nr: %u. Data saturated: %s. (%u-%u mm): %s\n",
195  (unsigned int)reflection_count,
196  (unsigned int)result_info.sequence_number,
197  result_info.data_saturated ? "true" : "false",
198  (unsigned int)(start_m * 1000.0f),
199  (unsigned int)(end_m * 1000.0f),
200  format_distances(reflection_count, reflections));
201 
202  format_distances(reflection_count, reflections);
203  }
204  else
205  {
206  fprintf(stderr, "Reflection data not properly retrieved\n");
207  return detector_status;
208  }
209  }
210 
211  detector_status = acc_detector_distance_peak_deactivate(handle);
212 
213  return detector_status;
214 }

217 char *format_distances(uint16_t reflection_count,
218  const acc_detector_distance_peak_reflection_t *reflections)
219 {
220  static char buffer[1024];
221  size_t total_count = 0;
222  int count;
223 
224  *buffer = 0;
225  for (uint_fast8_t reflection_index = 0; reflection_index < reflection_count; reflection_index++)
226  {
227  if (total_count > 0)
228  {
229  count = snprintf(&buffer[total_count], sizeof(buffer) - total_count, ", ");
230  if (count < 0)
231  {
232  break;
233  }
234 
235  total_count += count;
236  }
237 
238  count = snprintf(&buffer[total_count], sizeof(buffer) - total_count, "%u mm (%u)",
239  (unsigned int)((reflections[reflection_index].distance) * 1000.0f),
240  (unsigned int)(reflections[reflection_index].amplitude));
241  if (count < 0)
242  {
243  break;
244  }
245 
246  total_count += count;
247 
248  if (total_count >= sizeof(buffer) - 1)
249  {
250  break;
251  }
252  }
253 
254  return buffer;
255 }
