15 /**

17  *
18  * This is an example on how the envelope service can be used.

20  * - Initiate HAL layer
21  * - Activate Radar System Software (RSS)
22  * - Create a distance basic detector

98 bool parse_input(int argc, char *argv[], input_t *input)
99 {
100  input->count = 0;

102  input->range_start = 0.2f;

105  static struct option long_options[] =
106  {
107  {"count", required_argument, 0, 'c'},
108  {"range-start", required_argument, 0, 'r'},
109  {"range-length", required_argument, 0, 'l'},
110  {"sensor", required_argument, 0, 's'},

113  };

150  fprintf(stderr,

152  fprintf(stderr,

154  fprintf(stderr, "-s, --sensor The position where the sensor is connected, default 1\n");
